async function send() {
  const text = input.value;
  input.value = "";

  addMessage("You: " + text);

  const memory = await Walrus.read();

  let response = generateResponse(text, memory);

  addMessage("Agent: " + response);

  await Walrus.write({
    input: text,
    response,
    timestamp: Date.now()
  });
}

function generateResponse(input, memory) {
  const lower = input.toLowerCase();

  let past = "";

  if (memory.length > 0) {
    const last = memory[memory.length - 1];
    past = `You previously said: "${last.input}"`;
  }

  if (lower.includes("argentina")) {
    return `Interesting. You support Argentina. ${past}`;
  }

  if (lower.includes("brazil")) {
    return `Brazil noted. ${past}`;
  }

  return `Got it. ${past}`;
}

function detectPrediction(text) {
  const t = text.toLowerCase();

  if (t.includes("argentina")) return "Argentina";
  if (t.includes("brazil")) return "Brazil";
  if (t.includes("france")) return "France";
  if (t.includes("germany")) return "Germany";

  return null;
}

function generateResponse(input, memory) {
  const lower = input.toLowerCase();

  const predictions = memory
    .filter(m => m.type === "prediction")
    .map(m => m.value);

  let last = predictions[predictions.length - 1];

  // 1. prediction input
  const predictedTeam = detectPrediction(input);

  if (predictedTeam) {
    return `Noted. You predict ${predictedTeam} will win the World Cup.`;
  }

  // 2. query memory
  if (lower.includes("who did i predict")) {
    if (!last) return "You haven't made any predictions yet.";
    return `Your latest prediction is ${last}.`;
  }

  // 3. behavior change based on memory size
  if (memory.length >= 3) {
    return `You have made ${memory.length} entries. Your opinions are becoming inconsistent. Latest: ${last || "none"}`;
  }

  // default
  return "I recorded your input. Continue making predictions.";
}

async function send() {
  const text = input.value;
  if (!text) return;

  input.value = "";

  addMsg("You: " + text, "user");

  const memory = await Walrus.read();

  const predictedTeam = detectPrediction(text);

  if (predictedTeam) {
    await Walrus.write({
      type: "prediction",
      value: predictedTeam,
      time: Date.now()
    });
  } else {
    await Walrus.write({
      type: "message",
      value: text,
      time: Date.now()
    });
  }

  const updatedMemory = await Walrus.read();

  const reply = generateResponse(text, updatedMemory);

  addMsg("Agent: " + reply, "agent");
}

👛 4. wallet.js（Sui Wallet 登录）
async function connectWallet() {

  if (!window.suiWallet) {
    alert("Install Sui Wallet extension");
    return;
  }

  const res = await window.suiWallet.requestPermissions();

  window.walletAddress = res.accounts[0];

  document.getElementById("wallet").innerText =
    "Connected: " + window.walletAddress;

  return window.walletAddress;
}
🧠 5. memwal.js（真实 Memory Layer）
const MemWal = {

  async write(agentId, data) {
    return fetch(CONFIG.MEMWAL_ENDPOINT + "/write", {
      method: "POST",
      headers: {"Content-Type": "application/json"},
      body: JSON.stringify({
        agent_id: agentId,
        payload: data
      })
    });
  },

  async read(agentId) {
    const res = await fetch(
      CONFIG.MEMWAL_ENDPOINT + "/read?agent_id=" + agentId
    );
    return res.json();
  }
};

function mascot(memory) {

  if (!memory.length) {
    return "🐘 I am Walrus. Make your first prediction.";
  }

  const last = memory[memory.length - 1];

  if (last.payload?.text?.includes("Argentina")) {
    return "🐘 Argentina again... interesting pattern.";
  }

  if (memory.length > 3) {
    return "🐘 I remember everything... you are evolving.";
  }

  return "🐘 Continue... I am learning your football destiny.";
}
🤖 8. app.js（核心逻辑）
async function send() {

  const input = document.getElementById("input").value;

  const agentId = window.walletAddress;

  const entry = {
    type: "prediction",
    text: input,
    time: Date.now()
  };

  await MemWal.write(agentId, entry);

  const memory = await MemWal.read(agentId);

  render(memory);

  document.getElementById("mascotText").innerText =
    mascot(memory);
}

function render(memory) {

  const box = document.getElementById("memoryBox");
  box.innerHTML = "";

  memory.forEach(m => {
    const div = document.createElement("div");
    div.innerText = JSON.stringify(m.payload);
    box.appendChild(div);
  });
}
